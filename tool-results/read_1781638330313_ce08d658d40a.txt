    38→// Dynamically get a Lucide icon by name string
    39→function getIcon(iconName?: string): LucideIcon {
    40→  if (!iconName) return Construction;
    41→  const Icon = (LucideIcons as Record<string, LucideIcon>)[iconName];
    42→  return Icon || Construction;
    43→}
